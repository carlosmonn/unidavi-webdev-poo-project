13  public class Conta {
14  	private String agencia;
15  	private String numero;
16  	private double saldo;
17  	private StringBuilder extrato = new StringBuilder();
18  
19  	public String getAgencia() {
20  		return agencia;
21  	}
22  
23  	public void setAgencia(String agencia) {
24  		this.agencia = agencia;
25  	}
26  
27  	public String getNumero() {
28  		return numero;
29  	}
30  
31  	public void setNumero(String numero) {
32  		this.numero = numero;
33  	}
34  
35  	public double getSaldo() {
36  		return saldo;
37  	}
38  
39  	public void setSaldo(double saldo) {
40  		this.saldo = saldo;
41  	}
42  
43  	public void addDeposito(double valor) {
44  		movimento(valor, "Depósito");
45  	}
46  
47  	public String addSaque(double valor) {
48  		if (valor <= this.saldo) {
49  			movimento(-valor, "Saque");
50  			return "Saque realizado com sucesso!";
51  		}
52  
53  		return "Saldo insuficiente!";
54  	}
55  
56  	protected void movimento(double valor, String descricao) {
57  		this.saldo += valor;
58  
59  		extrato.append(descricao);
60  		extrato.append(" de: ");
61  		extrato.append(Math.abs(valor));
62  		extrato.append("\n");
63  	}
64  
65  	public String getExtrato() {
66  		StringBuilder impExtrato = new StringBuilder();
67  		impExtrato.append("\n");
68  		impExtrato.append("Agência: ");
69  		impExtrato.append(this.agencia);
70  		impExtrato.append("\n");
71  		impExtrato.append("Conta: ");
72  		impExtrato.append(this.numero);
73  		impExtrato.append("\n");
74  		impExtrato.append(this.extrato);
75  		impExtrato.append("\n");
76  		impExtrato.append("Saldo: ");
77  		impExtrato.append(this.saldo);
78  		impExtrato.append("\n");
79  
80  		return impExtrato.toString();
81  	}
82  }
