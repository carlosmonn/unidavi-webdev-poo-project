13  public class Corrente extends Conta {
14  	private double limite;
15  
16  	public double getLimite() {
17  		return limite;
18  	}
19  
20  	public void setLimite(double limite) {
21  		this.limite = limite;
22  	}
23  
24  	@Override
25  	public String addSaque(double valor) {
26  		if ((valor <= this.getSaldo()) || (valor <= (this.getSaldo() + this.getLimite()))) {
27  			movimento(-valor, "Saque");
28  			return "Saque realizado com sucesso!";
29  		}
30  
31  		return "Saldo insuficiente!";
32  	}
33  
34  	@Override
35  	public String getExtrato() {
36  		StringBuilder impExtrato = new StringBuilder();
37  		impExtrato.append("------- EXTRATO - CONTA CORRENTE -------");
38  		impExtrato.append(super.getExtrato());
39  		impExtrato.append("Limite: ");
40  		impExtrato.append(this.limite);
41  		impExtrato.append("\n");
42  		impExtrato.append("Limite dispon√≠vel: ");
43  		impExtrato.append(this.getLimiteDisponivel());
44  		impExtrato.append("\n");
45  		impExtrato.append("----------------------------------------");
46  
47  		return impExtrato.toString();
48  	}
49  
50  	public double getLimiteDisponivel() {
51  		if (this.getSaldo() < 0) {
52  			return this.getSaldo() + this.limite;
53  		}
54  
55  		return this.limite;
56  	}
57  }
